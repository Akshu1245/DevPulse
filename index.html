import { AuthService, PaymentGateway, OrderManager } from '@devpulse/api';

class IntegrationService {
  private auth: AuthService;
  private payments: PaymentGateway;
  private orders: OrderManager;

  constructor() {
    this.auth = new AuthService();
    this.payments = new PaymentGateway();
    this.orders = new OrderManager();
  }

  async processUserTransaction(userData: UserData, paymentDetails: PaymentInfo): Promise {
    // Validate user credentials
    const userToken = await this.auth.login(userData);

    // Create payment transaction
    const paymentResult = await this.payments.processPayment(paymentDetails, userToken);

    // Create order if payment successful
    if (paymentResult.success) {
      const order = await this.orders.createOrder(paymentResult.transactionId, userData);
      
      return {
        success: true,
        orderId: order.id,
        transactionId: paymentResult.transactionId
      };
    }

    return {
      success: false,
      error: paymentResult.error
    };
  }
}

export default IntegrationService;
